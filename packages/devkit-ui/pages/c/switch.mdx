```tsx
const [isChecked, setIsChecked] = useState(false)

  />
  <Switch.Toggle
    checked={isChecked}
    onChange={e => setIsChecked(e.target.checked)}
    className={isChecked ? 'text-green-400' : 'text-gray-300'}
    focusStyle={appContext.theme === 'dark' ? 'ring-offset-gray-800' : ''}
  />
  <Switch.Toggle
    isRect
    checked={isChecked}
    onChange={e => setIsChecked(e.target.checked)}
    className="rounded text-blue-400"
    focusStyle={appContext.theme === 'dark' ? 'ring-offset-gray-800' : ''}
  />
  <Switch.Toggle
    checked={isChecked}
    onChange={e => setIsChecked(e.target.checked)}
    className="text-purple-400"
    focusStyle={appContext.theme === 'dark' ? 'ring-offset-gray-800' : ''}
  >
    <Switch.On className="text-gray-800">on</Switch.On>
    <Switch.Off className="text-gray-800">off</Switch.Off>
  </Switch.Toggle>
</div>
```

### #With Label

<a id="scroll-to-switch--with-label" />

import { SwitchWithLabel } from 'examples/switch.example'

<SwitchWithLabel />
<a id="scroll-to-switch--with-label-code" />

```tsx
const appContext = useContext(AppContext)
<Switch>
  <Switch.Toggle
    checked={appContext.theme === 'dark'}
    onChange={() => {
      if (appContext.setTheme)
        appContext.setTheme(appContext.theme === 'dark' ? 'light' : 'dark')
    }}
    className={
      appContext.theme === 'dark' ? 'text-gray-600' : 'text-gray-300'
    }
    focusStyle={appContext.theme === 'dark' ? 'ring-offset-gray-800' : ''}
  />
  <Switch.Label>
    {appContext.theme} : you can also click me to change theme
  </Switch.Label>
</Switch>
```

### #API

<a id="scroll-to-switch--api" />

```tsx
// wrapping with `Switch` will provide
// `id` to use with `Toggle` and `Label` via context
// internally use `FormControl` component
<Switch></Switch>
// `Toggle` switch component which can also use as stand alone
<Switch.Toggle/>
interface ToggleProps {
  isRect?: boolean // `true` to make controller btn rectangle, default `false`
  focusStyle?: string // classname prop that will apply when `Toggle` is focused
}
// `Label` component can only use inside
// `Switch` component together with `Toggle`
<Switch.Label></Switch.Label>
// `On` component can only use inside
// `Toggle` component together with `Off`
<Switch.On></Switch.On>
// `Off` component can only use inside
// `Toggle` component together with `On`
<Switch.Off></Switch.Off>
```
