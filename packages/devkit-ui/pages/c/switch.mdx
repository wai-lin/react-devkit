# Switch

### #Sample

<SwitchSample />

```tsx
const [isChecked, setIsChecked] = useState(false)
<Switch.Toggle
  checked={isChecked}
  onChange={e => setIsChecked(e.target.checked)}
  className="text-green-500"
/>
<p>isChecked : {isChecked.toString()}</p>
```

### #Styling

<SwitchStyling />

```tsx
const [isChecked, setIsChecked] = useState(false)
const appContext = useContext(AppContext)
<p className="m-0">isChecked: {isChecked.toString()}</p>
<div className="flex items-center space-x-2">
  <Switch.Toggle
    checked={isChecked}
    onChange={e => setIsChecked(e.target.checked)}
    className="text-red-400"
  />
  <Switch.Toggle
    checked={isChecked}
    onChange={e => setIsChecked(e.target.checked)}
    className="text-yellow-400"
  />
  <Switch.Toggle
    checked={isChecked}
    onChange={e => setIsChecked(e.target.checked)}
    className="text-green-400"
    focusStyle={appContext.theme === 'dark' ? 'ring-offset-gray-800' : ''}
  />
</div>
```

### #With Label

<SwitchWithLabel />

```tsx
const appContext = useContext(AppContext)
<Switch>
  <Switch.Toggle
    checked={appContext.theme === 'dark'}
    onChange={() => {
      if (appContext.setTheme)
        appContext.setTheme(appContext.theme === 'dark' ? 'light' : 'dark')
    }}
    className="text-gray-600"
    focusStyle={appContext.theme === 'dark' ? 'ring-offset-gray-800' : ''}
  />
  <Switch.Label>
    {appContext.theme} : you can also click me to change theme
  </Switch.Label>
</Switch>
```
