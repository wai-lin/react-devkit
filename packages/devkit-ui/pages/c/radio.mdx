```tsx
const [isSelected, setIsSelected] = useState(false)
<Radio.Input id="radio-id" />
<label htmlFor="radio-id">select me</label>
<Radio>
  <Radio.Input
    checked={isSelected}
    onChange={e => setIsSelected(e.target.checked)}
  />
  <Radio.Label>radio selected : {isSelected.toString()}</Radio.Label>
</Radio>
```

### #Group

<a id="scroll-to-radio--group" />

import { RadioGrouping } from 'examples/radio.example'

<RadioGrouping />
<a id="scroll-to-radio--group-code" />

```tsx
const [selected, setSelected] = useState('')
const options = ['HTML', 'CSS', 'JS']
<p>selected option : {selected}</p>
<Radio.Group>
  {options.map(op => (
    <Radio key={op}>
      <Radio.Input
        checked={selected === op}
        value={op}
        onChange={e => setSelected(e.target.value)}
      />
      <Radio.Label>{op}</Radio.Label>
    </Radio>
  ))}
</Radio.Group>
```
