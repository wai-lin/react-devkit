### #API

<a id="scroll-to-otp-input--api" />

```tsx
// wrapper component for `OtpInput`
// provide otp functionaly context
;<OtpInput.Group></OtpInput.Group>
interface OtpGroupProps {
  value: string // required
  setValue: Dispatch<SetStateAction<string>> // required
  acceptSpaceKey?: boolean // set `true` to accept space as value default  `false`
  validator?: (value: string) => boolean // validator function to check each input, must return `true` | `false`
}
// Input component to use inside `Group`
;<OtpInput />
interface OtpInputProps {
  index: number // required, `OtpInput` depends on this prop, must starts from `0`
  disabled?: boolean // whether input is disabled or not
  type?: 'text' | 'number' | 'password' // input type
}
```
